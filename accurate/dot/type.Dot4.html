type Dot4<F> = DotK<F, Sum3<F>>;

DotK with four cascaded accumulators

Examples

use accurate::traits::*;
use accurate::dot::Dot4;

let d = Dot4::zero() + (1.0, 1.0) + (2.0, 2.0) + (3.0, 3.0);
assert_eq!(14.0f64, d.dot());
